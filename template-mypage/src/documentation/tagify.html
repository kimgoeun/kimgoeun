Tagify

Lightweight input "tags" script.

Tagify documentation

How to use

Copy-paste the stylesheet <link> into your <head> to load the CSS.

              
                <link rel="stylesheet" href="@@autopath/node_modules/@yaireo/tagify/dist/tagify.css">

Copy-paste the following <script> near the end of your pages under JS Implementing Plugins to enable it.

              
                <script src="@@autopath/node_modules/@yaireo/tagify/dist/tagify.min.js"></script>

Copy-paste the following <script> near the end of your pages under JS Front to enable it.

              
                <script src="@@autopath/assets/js/hs.tagify.js"></script>

Copy-paste the init function under JS Plugins Init., before the closing </body> tag, to enable it.

              
                <script>
                  $(document).on('ready', function () {
                    // INITIALIZATION OF TAGIFY
                    // =======================================================
                    $('.js-tagify').each(function () {
                      var tagify = $.HSCore.components.HSTagify.init($(this));
                    });
                  });
                </script>

Blacklist

Use "blacklist": ["apple", "ananas", ...] prop.

  • Preview
  • HTML 
                    
                      <!-- Tagify -->
                      <input type="text" class="js-tagify tagify-form-control form-control" placeholder="Write some tags" value="new, bestseller, apple"
                             data-hs-tagify-options='{
                               "blacklist": ["apple", "ananas"]
                             }'>

Remove Button

Use "clearBtnSelector": "#removeTagsBtn" prop.

  • Preview
  • HTML 
                    
                      <!-- Tagify -->
                      <div class="form-group">
                        <input type="text" class="js-tagify tagify-form-control form-control" placeholder="Write some tags" value="css, html, javascript"
                               data-hs-tagify-options='{
                                 "clearBtnSelector": "#removeTagsBtn"
                               }'>

                        <button id="removeTagsBtn" class="btn btn-primary">Remove all</button>
                      </div>
                      <!-- End Tagify -->

                      <script>
                       $('.js-tagify-avatars').each(function () {
                         var settings = $(this).attr('data-hs-tagify-options') ? JSON.parse($(this).attr('data-hs-tagify-options')) : {},
                           tagifyAvatars = $.HSCore.components.HSTagify.init($(this), {
                             templates: {
                               tag: function tag(tagData) {
                                 try {
                                  return "<tag title=\"" + tagData.value + "\" contenteditable=\"false\" spellcheck=\"false\" class=\"tagify__tag " + (tagData["class"] ? tagData["class"] : "") + "\" " + this.getAttributes(tagData) + ">" +
                                            "<x title=\"remove tag\" class=\"tagify__tag__removeBtn\"></x>" +
                                            "<div class=\"d-flex align-items-center\">" +
                                            "" + (tagData.src ? "<img class=\"avatar avatar-xss avatar-circle mr-2\" src=\"" + tagData.src.toLowerCase() + "\">" : "") + "" +
                                              "<span>" + tagData.value + "</span>" +
                                            "</div>" +
                                          "</tag>";
                                 } catch (err) {}
                               },
                               dropdownItem: function dropdownItem(tagData) {
                                try {
                                  return "<div " + this.getAttributes(tagData) + " class=\"tagify__dropdown__item " + (tagData["class"] ? tagData["class"] : "") + "\">" +
                                           "<img class=\"avatar avatar-xss avatar-circle mr-2\" src=\"" + tagData.src.toLowerCase() + "\">" +
                                           "<span>" + tagData.value + "</span>" +
                                         "</div>";
                                } catch (err) {}
                               }
                             }
                           }).addTags(settings.whitelist.slice(0, 1));
                         });
                       });
                      </script>
