                      <script>
                        $(document).on('ready', function () {
                          // INITIALIZATION OF TOASTS
                          // =======================================================
                          $('.toast-show').toast({
                            autohide: false
                          });

                          $('.toast-show').toast('show');

                          $('#toast-invoker').click(function () {
                            $('#toast').toast({
                              delay: 1500
                            });

                            $('#toast').toast('show');
                          });
                        });
                      </script>
